# 八皇后问题

这里使用回溯的思想去解决八皇后问题,当然八皇后的问题变种有很多,这里还是需要我们深入的思考.

````java
/**
 * @author jbzm
 * @date 2019-08-13 15:37
 */
public class EightQueen {

  private static int[] queue = new int[8];
  private static int a = 0;

  private static boolean check(int currentPoint) {
    for (int j = currentPoint - 1; j >= 0; j--) {
      if (queue[j] == queue[currentPoint]) {
        return false;
      }
      if (Math.abs(currentPoint - j) == Math.abs(queue[currentPoint] - queue[j])) {
        return false;
      }
    }
    return true;
  }

  private static boolean goBack(int element) {
    if (element == 8) {
      a++;
      System.out.println(Arrays.toString(queue));
      return true;
    }
    for (int i = 0; i < 8; i++) {
      queue[element] = i;
      if (check(element)) {
        goBack(element + 1);
      }
    }
    return false;
  }

  @Test
  public void test() {
    goBack(0);
    System.out.println(a);
  }
}
````
